  <script>
    // Elements
    const openBtn   = document.getElementById('openSignIn');
    const modal     = document.getElementById('signInModal');
    const overlay   = document.getElementById('modalOverlay');
    const closeBtn  = document.getElementById('closeSignIn');
    const cancelBtn = document.getElementById('cancelBtn');
    const form      = document.getElementById('signinForm');
    const username  = document.getElementById('username');
    const password  = document.getElementById('password');
    const togglePwd = document.getElementById('togglePwd');
    const formStatus= document.getElementById('formStatus');

    // Helpers
    const openModal = () => {
      modal.classList.add('active');
      overlay.classList.add('active');
      modal.setAttribute('aria-hidden', 'false');
      overlay.setAttribute('aria-hidden', 'false');
      // Focus username after the animation frame
      requestAnimationFrame(() => username.focus());
      // Close on overlay click
      overlay.addEventListener('click', closeModal, { once:true });
      document.addEventListener('keydown', escToClose);
    };

    const closeModal = () => {
      modal.classList.remove('active');
      overlay.classList.remove('active');
      modal.setAttribute('aria-hidden', 'true');
      overlay.setAttribute('aria-hidden', 'true');
      document.removeEventListener('keydown', escToClose);
      openBtn.focus();
    };

    const escToClose = (e) => {
      if (e.key === 'Escape') closeModal();
    };

    // Toggle password visibility
    togglePwd.addEventListener('click', () => {
      const isHidden = password.type === 'password';
      password.type = isHidden ? 'text' : 'password';
      togglePwd.setAttribute('aria-label', isHidden ? 'Hide password' : 'Show password');
      togglePwd.innerHTML = isHidden ? '<i class="far fa-eye-slash"></i>' : '<i class="far fa-eye"></i>';
      password.focus();
    });

    // Demo open
    openBtn.addEventListener('click', openModal);
    closeBtn.addEventListener('click', closeModal);
    cancelBtn.addEventListener('click', closeModal);

    // Basic client-side validation + placeholder auth hook
    form.addEventListener('submit', (e) => {
      e.preventDefault();
      formStatus.textContent = ''; // reset

      const user = username.value.trim();
      const pass = password.value;

      if (!user || !pass) {
        formStatus.textContent = 'Please enter both a username and password.';
        // focus the first empty field
        (!user ? username : password).focus();
        return;
      }

      if (pass.length < 8) {
        formStatus.textContent = 'Password must be at least 8 characters.';
        password.focus();
        return;
      }

// Track current user
let currentUser = null;

// Reference to user info div
const userInfo = document.getElementById('userInfo');

// Function to update UI based on login status
function updateUserDisplay() {
  if (currentUser) {
    // Show username and sign-out button
    userInfo.innerHTML = `
      Welcome, <strong>${currentUser}</strong>
      <br />
      <button id="signOutBtn" class="btn ghost" style="margin-top:8px;">Sign Out</button>
    `;

    // Hide the sign-in button since user is logged in
    openBtn.style.display = "none";

    // Add listener for sign-out
    document.getElementById('signOutBtn').addEventListener('click', () => {
      currentUser = null;
      updateUserDisplay();
    });
  } else {
    // No user logged in ‚Üí clear and show sign-in button
    userInfo.textContent = "";
    openBtn.style.display = "inline-block";
  }
}

currentUser = user;   // store username after sign-in
